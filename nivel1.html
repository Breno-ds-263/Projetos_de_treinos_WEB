Identifique as definições e os usos das variáveis no código abaixo:

def calculate_total(preco, taxa):
    total = preco + (preco * taxa)  
    return total 

def main():
    preco = 100  
    taxa = 0.05  
    result = calculate_total(preco, taxa) 
    print(result)  

main()
